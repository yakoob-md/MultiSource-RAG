CREATE: backend/rag/multi_retriever.py

IMPORTS: from backend.rag.retriever import retrieve, RetrievedChunk
         from backend.rag.query_classifier import QueryAnalysis
         from backend.database.connection import get_connection
         import json

DEFINE DATACLASS: MultiSourceResult
    query_intent     : str
    source_groups    : dict[str, list[RetrievedChunk]]  # source_title -> chunks
    all_chunks       : list[RetrievedChunk]
    source_count     : int

DEFINE FUNCTION: retrieve_single_source(question, source_ids, top_k=8) -> MultiSourceResult
    Call retrieve(question, source_ids) once.
    Group result by source_title.
    Return MultiSourceResult(query_intent="single_source", source_groups={...}, all_chunks=[...], source_count=1)

DEFINE FUNCTION: retrieve_for_comparison(question, analysis: QueryAnalysis, top_k_per_source=5) -> MultiSourceResult
    This must retrieve independently from EACH source type in analysis.source_types.
    For each source_type:
        1. Query MySQL: SELECT id FROM sources WHERE ... (use source type + name matching)
        2. Call retrieve(question, source_ids=[those ids], top_k_per_source)
        3. Store as source_groups[source_title]
    If a source type returns 0 results, log warning but continue.
    Return MultiSourceResult with all groups populated.
    IMPORTANT: Do NOT merge chunks from different sources before returning.
    Keep them grouped so generator can compare them independently.

DEFINE FUNCTION: retrieve_for_synthesis(question, top_k=16) -> MultiSourceResult
    Call retrieve(question, source_ids=None, top_k=top_k) — broad search across all.
    Group by source_title.
    Cap each source at top_k // len(unique_sources) chunks to ensure balance.
    Return MultiSourceResult with balanced source_groups.

DEFINE FUNCTION: multi_retrieve(question: str, analysis: QueryAnalysis) -> MultiSourceResult
    Route based on analysis.intent:
    - "single_source" -> retrieve_single_source(question, source_ids from analysis)
    - "comparison"    -> retrieve_for_comparison(question, analysis)
    - "synthesis"     -> retrieve_for_synthesis(question)
    Log: print(f"[MultiRetriever] intent={analysis.intent} | sources={analysis.source_types}")

    Validation checklist:

Single source query returns source_count == 1
Comparison query returns source_count >= 2 with distinct source_title keys in source_groups
Synthesis query returns balanced chunks (no single source dominates >60% of results)
All three functions handle empty retrieval results without raising exceptions
