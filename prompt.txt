  export async function streamQueryRag(
    question: string,
    sourceIds: string[] | undefined,
    history: Array<{role: string; content: string}>,
    onToken: (token: string) => void,
    onDone: (citations: Citation[]) => void,
    onError: (err: Error) => void
  ): Promise<void> {
    try {
      const res = await fetch(`${API_BASE}/query/stream`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          question,
          source_ids: sourceIds && sourceIds.length > 0 ? sourceIds : null,
          history,
        }),
      })

      if (!res.ok) throw new Error(`HTTP ${res.status}`)
      if (!res.body) throw new Error('No response body')

      const reader = res.body.getReader()
      const decoder = new TextDecoder()

      while (true) {
        const { done, value } = await reader.read()
        if (done) break

        const text = decoder.decode(value, { stream: true })
        const lines = text.split('\n')

        for (const line of lines) {
          if (!line.startsWith('data: ')) continue
          const jsonStr = line.slice(6).trim()
          if (!jsonStr) continue

          try {
            const parsed = JSON.parse(jsonStr)
            if (parsed.error) {
              onError(new Error(parsed.error))
              return
            }
            if (parsed.done) {
              onDone(parsed.citations ?? [])
              return
            }
            if (parsed.token) {
              onToken(parsed.token)
            }
          } catch {
            // partial chunk — skip silently
          }
        }
      }
    } catch (err) {
      onError(err instanceof Error ? err : new Error(String(err)))
    }
  }

  IMPORTANT: Citation type must match existing Citation interface in types.ts.
  If Citation is not yet imported in api.ts, add the import from types.ts.

1. Groq SDK version:
   Check groq package version in requirements.txt.
   stream=True requires groq >= 0.4.0
   If version is lower: pip install --upgrade groq

2. CORS with SSE:
   In main.py, ensure CORSMiddleware allows_origins includes
   the frontend dev URL (typically http://localhost:5173).
   If CORS is blocking SSE, add these allowed headers:
   ["Content-Type", "Cache-Control", "X-Accel-Buffering"]

3. Generator.py import:
   If _build_prompt() or the citation builder are not standalone
   importable functions (e.g., buried inside a class), refactor
   only that function to be importable — do not touch any other
   logic in generator.py.

4. Empty stream from Groq:
   If Groq returns 0 tokens (content filter triggered), the stream
   ends without a done event. Add a post-loop check:
   if not sent_done_event: yield done event with empty citations

5. Windows line endings:
   SSE requires \n\n line endings. On Windows, ensure you're not
   accidentally producing \r\n. Use explicit "\n\n" strings.
